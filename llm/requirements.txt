# Install into the `qwen-vllm` conda env (Python 3.12):
#   conda activate qwen-vllm
#   pip install -r llm/requirements.txt \
#       --extra-index-url https://download.pytorch.org/whl/cu128 \
#       -c llm/constraints-cu128.txt
#
# Why pinned:
#  - The local NVIDIA driver (570.x) caps at CUDA 12.8, so torch/vLLM must be
#    cu128 builds — the latest vLLM (0.22+) targets CUDA 13 and will NOT run here.
#  - vLLM 0.11.2 pins torch==2.9.0 and ships the xformers attention backend
#    (flashinfer optional), which is what fits this 8 GB Ada GPU.
#  - On an ecryptfs home, flashinfer's cubin filenames are too long; see README
#    (symlink flashinfer dirs onto /var/tmp before installing).
vllm==0.11.2
openai>=1.40.0
